DROP TABLE IF EXISTS `cm_board`;
CREATE TABLE `cm_board` (
  `board_num` int(11) NOT NULL AUTO_INCREMENT,
  `group_id` varchar(100) NOT NULL,
  `board_id` varchar(100) NOT NULL,
  `parent_num` int(11) DEFAULT NULL,
  `reply_depth` int(11) NOT NULL,
  `reply_order` int(11) NOT NULL,
  `user_id` varchar(255) NOT NULL,
  `email` varchar(100) NOT NULL,
  `name` varchar(50) NOT NULL,
  `password` varchar(255) NOT NULL,
  `title` varchar(255) NOT NULL,
  `content` text NOT NULL,
  `ip` varchar(255) NOT NULL,
  `view_count` int(11) DEFAULT '0',
  `reg_date` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_date` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`board_num`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `cm_board_comment`;
CREATE TABLE `cm_board_comment` (
  `comment_id` int(11) NOT NULL AUTO_INCREMENT,
  `board_id` varchar(100) NOT NULL,
  `board_num` int(11) NOT NULL,
  `user_id` varchar(255) DEFAULT NULL,
  `name` varchar(50) NOT NULL,
  `email` varchar(100) NOT NULL,
  `password` varchar(255) NOT NULL,
  `content` text NOT NULL,
  `ip` varchar(255) NOT NULL,
  `reg_date` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_date` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`comment_id`),
  KEY `board_num` (`board_num`),
  CONSTRAINT `cm_board_comment_ibfk_1` FOREIGN KEY (`board_num`) REFERENCES `cm_board` (`board_num`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `cm_board_file`;
CREATE TABLE `cm_board_file` (
  `file_id` int(11) NOT NULL AUTO_INCREMENT,
  `board_id` varchar(255) NOT NULL,
  `board_num` int(11) NOT NULL,
  `original_filename` varchar(255) NOT NULL,
  `stored_filename` varchar(255) NOT NULL,
  `file_size` int(11) NOT NULL,
  `file_type` varchar(100) NOT NULL,
  `reg_date` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`file_id`),
  KEY `board_id` (`board_num`),
  CONSTRAINT `cm_board_file_ibfk_1` FOREIGN KEY (`board_num`) REFERENCES `cm_board` (`board_num`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `cm_board_group`;
CREATE TABLE `cm_board_group` (
  `group_id` varchar(100) NOT NULL,
  `group_name` varchar(255) NOT NULL DEFAULT '',
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`group_id`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `cm_board_list`;
CREATE TABLE `cm_board_list` (
  `board_id` varchar(255) NOT NULL,
  `board_name` varchar(255) NOT NULL,
  `group_id` varchar(100) NOT NULL,
  `group_name` varchar(100) NOT NULL,
  `board_skin` varchar(50) NOT NULL DEFAULT 'basic',
  `write_lv` int(11) NOT NULL,
  `list_lv` int(11) NOT NULL,
  `view_lv` int(11) NOT NULL,
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`board_id`),
  UNIQUE KEY `board_name` (`board_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

  `admin_id` varchar(255) NOT NULL,
  `site_title` varchar(255) NOT NULL,
  `admin_email` varchar(255) NOT NULL,
  `contact_number` varchar(50) DEFAULT NULL,

  `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


DROP TABLE IF EXISTS `cm_content`;
CREATE TABLE `cm_content` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `co_id` varchar(20) NOT NULL DEFAULT '',
  `co_subject` varchar(255) NOT NULL DEFAULT '',
  `co_content` longtext NOT NULL,

DROP TABLE IF EXISTS `cm_menu`;
CREATE TABLE `cm_menu` (
  `menu_id` int(11) NOT NULL AUTO_INCREMENT,
  `parent_id` int(11) DEFAULT '0',
  `menu_name` varchar(100) NOT NULL,
  `menu_url` varchar(255) DEFAULT '',
  `menu_icon` varchar(255) DEFAULT NULL,
  `target_blank` tinyint(1) DEFAULT '0',
  `is_disabled` tinyint(1) DEFAULT '0',
  `menu_level` int(11) NOT NULL COMMENT '메뉴의 중첩 레벨 (1부터 무제한)',
  `sort_order` int(11) DEFAULT '0' COMMENT '동일 상위 메뉴 내 출력 순서',
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`menu_id`),
  KEY `idx_parent_id` (`parent_id`),
  KEY `idx_menu_level` (`menu_level`),
  KEY `idx_parent_sort` (`parent_id`,`sort_order`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

DROP TABLE IF EXISTS `cm_popup`;
CREATE TABLE `cm_popup` (
  `po_id` int(11) NOT NULL AUTO_INCREMENT,
  `po_title` varchar(255) NOT NULL COMMENT '팝업제목',
  `po_content` text COMMENT '팝업내용',
  `po_top` int(11) NOT NULL DEFAULT '0' COMMENT '상단위치',
  `po_left` int(11) NOT NULL DEFAULT '0' COMMENT '좌측위치',
  `po_width` int(11) NOT NULL DEFAULT '0' COMMENT '가로사이즈',
  `po_height` int(11) NOT NULL DEFAULT '0' COMMENT '세로사이즈',
  `po_start_date` date DEFAULT NULL COMMENT '시작일',
  `po_end_date` date DEFAULT NULL COMMENT '종료일',
  `po_cookie_time` int(11) NOT NULL DEFAULT '24' COMMENT '쿠키 유지시간(시간)',
  `po_url` varchar(255) DEFAULT NULL COMMENT '팝업 URL',
  `po_target` varchar(10) NOT NULL DEFAULT '_blank' COMMENT '타겟(_blank, _self)',
  `po_use` tinyint(4) NOT NULL DEFAULT '1' COMMENT '사용여부(1:사용, 0:미사용)',
  `po_reg_date` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '등록일',
  PRIMARY KEY (`po_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='팝업레이어 관리';


DROP TABLE IF EXISTS `cm_users`;

  `user_id` varchar(255) NOT NULL COMMENT '회원아이디',
  `user_name` varchar(255) NOT NULL COMMENT '이름',

  `user_block` int(11) NOT NULL COMMENT '회원차단1',
